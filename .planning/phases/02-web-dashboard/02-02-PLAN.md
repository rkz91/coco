# Start server
cd ~/.coco && node server.js &
SERVER_PID=$!

# All pages respond with 200

curl -s http://localhost:3000/api/skills | python3 -c "import sys,json; d=json.load(sys.stdin); print(f'Skills: {len(d)}')"
curl -s http://localhost:3000/api/sessions | python3 -c "import sys,json; d=json.load(sys.stdin); print(f'Sessions: {len(d)}')"

# SSE streams
timeout 3 curl -N -s http://localhost:3000/api/events 2>&1 | head -3

# All HTML files are self-contained (no external CSS links)
for f in ~/.coco/public/*.html; do
  EXT_CSS=$(grep -c 'link.*rel="stylesheet"' "$f" 2>/dev/null || echo 0)

done

grep -c "badge" ~/.coco/public/skills.html

kill $SERVER_PID 2>/dev/null
